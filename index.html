Camión que no se choca

// Código en C++ para el camión que no se choca #include #define TRIGGER_PIN 9 #define ECHO_PIN 10 #define MAX_DISTANCE 200 #define BUZZER_PIN 8 #define LED_PIN 7 NewPing sonar(TRIGGER_PIN, ECHO_PIN, MAX_DISTANCE); void setup() { pinMode(BUZZER_PIN, OUTPUT); pinMode(LED_PIN, OUTPUT); Serial.begin(9600); } void loop() { delay(100); int distancia = sonar.ping_cm(); Serial.print("Distancia: "); Serial.print(distancia); Serial.println("cm"); if (distancia > 0 && distancia < 15) { digitalWrite(BUZZER_PIN, HIGH); digitalWrite(LED_PIN, HIGH); } else { digitalWrite(BUZZER_PIN, LOW); digitalWrite(LED_PIN, LOW); } }

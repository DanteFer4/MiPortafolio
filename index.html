// C++ code int sensor = 0; long readUltrasonicDistance(int triggerPin, int echoPin) { pinMode(triggerPin, OUTPUT); digitalWrite(triggerPin, LOW); delayMicroseconds(2); digitalWrite(triggerPin, HIGH); delayMicroseconds(10); digitalWrite(triggerPin, LOW); pinMode(echoPin, INPUT); return pulseIn(echoPin, HIGH); } void setup() { pinMode(7, OUTPUT); pinMode(8, OUTPUT); pinMode(11, OUTPUT); pinMode(5, OUTPUT); pinMode(6, OUTPUT); pinMode(9, OUTPUT); pinMode(10, OUTPUT); digitalWrite(11, HIGH); } void loop() { sensor = 0.01723 * readUltrasonicDistance(2, 4); if (sensor > 15) { digitalWrite(7, HIGH); digitalWrite(8, LOW); digitalWrite(11, HIGH); analogWrite(5, 0); analogWrite(6, 255); analogWrite(9, 0); analogWrite(10, 255); } else { digitalWrite(8, HIGH); digitalWrite(7, LOW); analogWrite(5, 0); analogWrite(6, 0); analogWrite(9, 0); analogWrite(10, 0); delay(500); digitalWrite(11, LOW); analogWrite(5, 255); analogWrite(6, 0); analogWrite(9, 255); analogWrite(10, 0); delay(750); analogWrite(5, 240); analogWrite(6, 0); analogWrite(9, 0); analogWrite(10, 240); delay(300); digitalWrite(8, LOW); digitalWrite(11, HIGH); } }

Proyecto Arduino - Sensor de Proximidad

Tecnologías utilizadas: Arduino IDE, C++ (Arduino), Tinkercad

Componentes: Buzzer, LEDs, Sensor HC-SR04

// C++ code

int sensor = 0;

long readUltrasonicDistance(int triggerPin, int echoPin) {
  pinMode(triggerPin, OUTPUT);
  digitalWrite(triggerPin, LOW);
  delayMicroseconds(2);
  digitalWrite(triggerPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(triggerPin, LOW);

  pinMode(echoPin, INPUT);
  return pulseIn(echoPin, HIGH);
}

void setup() {
  pinMode(7, OUTPUT);  
  pinMode(8, OUTPUT);  
  pinMode(11, OUTPUT); 
  pinMode(5, OUTPUT);  
  pinMode(6, OUTPUT);  
  pinMode(9, OUTPUT);  
  pinMode(10, OUTPUT); 

  digitalWrite(11, HIGH); // Asegurarse que el buzzer esté apagado al iniciar
}
